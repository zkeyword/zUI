120 		if (hour) {

122 				expire = new Date();
123 			expire.setTime(today.getTime() + 36E5 * hour);
124 		}
125 		document.cookie = name + "=" + escape(value) + "; " + (hour ? "expires=" + expire.toGMTString() + "; " : "") + (path ? "path=" + path + "; " : "path=/; ") + (domain ? "domain=" + domain + ";" : "");
126 		return true;
127 	},
128 	get: function (name) {
129 		var r = new RegExp("(?:^|;+|\\s+)" + name + "=([^;]*)"),
130 			m = document.cookie.match(r);

132 	},
133 	del: function (name, domain, path) {
134 		document.cookie = name + "=; expires=Mon, 26 Jul 1997 05:00:00 GMT; " + (path ? "path=" + path + "; " : "path=/; ") + (domain ? "domain=" + domain + ";" : "");

199 /**
200  * 拖动插件 
201  * @constructor zUI.ui.tab
202  * @extends zUI.ui
203  * @requires zUI.base
204  * @author norion
205  * @blog http://zkeyword.com/
206  * @example 
207    zUI.ui.drag({
208 	   dragItem:'#dragItem',
209 	   dragWrap:'#dragWrap'
210    });
211  */
212 zUI.ui.drag = function(options){
213 	var o = options || {};
214 	if( !o.dragItem ) return false;
215 	var	dragItem = $('body').find(o.dragItem),
216 		dragWrap = $('body').find(o.dragWrap),
217 		win      = parent.document || document,
218 		mouse    = {x:0,y:0};
219 	function _moveDialog(event){
220         var e    = window.event || event,
221         	top  = parseInt(dragWrap.css('top')) + (e.clientY - mouse.y),
222         	left = parseInt(dragWrap.css('left')) + (e.clientX - mouse.x);
223         dragWrap.css({top:top,left:left});
224         mouse.x = e.clientX;
225         mouse.y = e.clientY;
226     };
227     dragItem.mousedown(function(event){
228         var e = window.event || event;
229         mouse.x = e.clientX;
230         mouse.y = e.clientY;
231         $(win).bind('mousemove', _moveDialog);
232     });
233     $(win).mouseup(function(event){
234         $(win).unbind('mousemove', _moveDialog);
235     });
236 };

239 /**
240  * ttabs选项卡插件
241  * @constructor zUI.ui.tab
242  * @extends zUI.ui
243  * @requires zUI.base
244  * @author norion
245  * @blog http://zkeyword.com/
246  * @example 
247 	zUI.ui.tab({
248 		tabItem:'#tab li',
249 		tabWrap:'#tab div',
250 		isAuto:false,
251 		autoTime:100,
252 		onclick:function(i){
253 			alert(i)
254 		}
255 	});
256  */
257 zUI.ui.tab = function(options){
258 	var o = options || {};
259 	if( !o.tabItem ) return false;
260 	var tabItem   = o.tabItem,                          //tab选卡对象
261 		tabWrap   = o.tabWrap || null,                  //tab切换内容对象
262 		tabEvent  = o.tabEvent || 'click',              //切换事件
263 		tabIndex  = o.tabIndex || 0,                    //初始位置
264 		isAuto    = o.isAuto || false,                  //是否自动播放
265 		autoTime  = o.autoTime || 2000,                 //自动播放时间
266 		autoSpeed = o.autoSpeed || 0,                   //自动播放速度
267 		onclick   = o.onclick ? o.onclick : null;       //切换后执行的函数
268 	
269 	/*切换动作*/
270 	var tabFn = {
271 		/*初始化*/
272 		init: function(){
273 			$(tabWrap).eq(tabIndex).show().siblings(tabWrap).hide();
274 			if( tabWrap != null ){
275 				var index = tabIndex;
276 				$(tabItem).bind(tabEvent,function(){
277 					index = $(tabItem).index(this);
278 					tabFn.cutoverFn(index);
279 				});
280 				if( isAuto ){
281 					tabFn.autoFn(index);
282 				}
283 			}
284 		},
285 		
286 		/*切换函数*/
287 		cutoverFn: function(i){
288 			//tab切换内容的html不为空才做下面动作
289 			if( $(tabWrap).eq(i).html() !== '' ){
290 				if( autoSpeed ){
291 					$(tabWrap).eq(i).stop(true,true).fadeIn(autoSpeed).siblings(tabWrap).fadeOut(autoSpeed);
292 				}else{
293 					$(tabWrap).eq(i).stop(true,true).show().siblings(tabWrap).hide();
294 				}
295 				$(tabItem).eq(i).addClass('on').siblings(tabItem).removeClass('on');
296 			}else{
297 				$(tabWrap).hide();
298 			}
299 			//点击tabItem执行函数
300 			if( zUI.base.isFunction(onclick) ){
301 				onclick(i);
302 			}
303 		},
304 		
305 		/*自动播放函数*/
306 		autoFn: function(i){
307 			var _mun    = $(tabWrap).size(),
308 				_MyTime = setInterval(function(){
309 					tabFn.cutoverFn(i);
310 					i++;
311 					if( i === _mun ){
312 						i = 0;
313 					};
314 				},autoTime);
315 			$(tabItem).parent().hover(function(){
316 				clearInterval(_MyTime);
317 			},function(){
318 				_MyTime = setInterval(function(){
319 					tabFn.cutoverFn(i);
320 					i ++;
321 					if( i === _mun ){
322 						i = 0;
323 					};
324 				},autoTime);
325 			});
326 		}
327 	};//end tabfn
328 	tabFn.init();
329 };

340 zUI.ui.tip = function(options){

342 	var id                = o.id || 'l-tip-'+(new Date()).valueOf(),
343 		tipItem           = $(o.tipItem),
344 		tipItemWidth      = tipItem[0].offsetWidth,
345 		tipItemHeight     = tipItem[0].offsetHeight,

349 		tipItemWrap       = $(o.tipItemWrap||'body'),        //tipItem的最外层，默认网页的最外层为body，因为有可能在其他元素中定位
350 		tipItemWrapWidth  = tipItemWrap[0].offsetWidth,
351 		tipItemWrapHeight = tipItemWrap[0].offsetHeight,
352 		text              = o.text || attrTitle,             //tip的内容
353 		width             = o.width || 150,

356 		isArrow           = o.isArrow || false,              //是否需要箭头

358 		isInitShow        = o.isInitShow || false,           //初始化显示tip
359 		event             = o.event || 'mouseover';          //触发显示tip
360 	
361 	if( !text ){return;}
362 	
363 	var tipFn = {
364 		init: function(){
365 			if( $('#'+id).length ){ return; }
366 			if( tipItem ){ tipItem.removeAttr('title'); }
367 			var h = '';
368 				h += '<div id="'+ id +'" class="l-ui l-tip">';
369 				h += '<div class="l-tipMain">';
370 				if( tipHeader ){
371 					h += '<h5>'+ tipHeader +'</h5>';

378 			$('#l-ui-wrap').prepend(h);

381 		},
382 		
383 		/*默认位置*/
384 		defaultPositon: function(){
385 			this.init();
386 			var tipWrap = $('#'+id).addClass('l-tip-default'),
387 				top     = 0,
388 				left    = 0;
389 			left = tipItemLeft;
390 			top = tipItemTop + tipItemHeight + 5;
391 			tipWrap.css({top:top,left:left});
392 		},
393 		
394 		/*鼠标跟随*/
395 		track: function(){
396 			this.init();
397 			var tipWrap       = $('#'+id).addClass('l-tip-track'),
398 				mousePosition = zUI.ui.mousePosition();
399 			tipWrap.css({top:mousePosition.positionY + 5,left:mousePosition.positionX  + 5});
400 		},
401 		
402 		/*带箭头*/
403 		arrow:function(){
404 			this.init();
405 			var tipWrap       = $('#'+id).addClass('l-tip-Arrow'),
406 				tipArrow      = tipWrap.prepend('<span class="l-tipArrow"></span>').find('.l-tipArrow'),
407 				tipWrapHeight = tipWrap[0].offsetHeight,
408 				tipWrapWidth  = tipWrap[0].offsetWidth,
409 				top           = 0,
410 				left          = 0;
411 			if( arrowDirection === 'topBottom' ){
412 				tipArrow.addClass('l-tipArrow-topBottom');
413 				var tipArrowH = tipArrow.height(),
414 					tipArrowW = tipArrow.width();
415 				/*判断obj在左还是在右*/
416 				if( tipItemLeft < tipItemWrapWidth/2 - tipItemWidth/2 ){
417 					left = tipItemLeft;
418 					/*判断obj在上还是在下*/
419 					if( tipItemTop < tipItemWrapHeight/2 - tipItemHeight/2 ){

426 				}else{
427 					left = tipItemLeft + tipItemWidth - tipWrapWidth;
428 					/*判断obj在上还是在下*/

430 						top = tipItemTop + tipItemHeight + tipArrowH;
431 						tipArrow.addClass('l-tipArrow-topRight');
432 					}else{
433 						top = tipItemTop - tipWrapHeight - tipArrowH;
434 						tipArrow.addClass('l-tipArrow-bottomRight');
435 					}
436 				}
437 			}else{
438 				var tipArrowH = tipArrow.height(),
439 					tipArrowW = tipArrow.width();
440 				tipArrow.addClass('l-tipArrow-leftRight');
441 				/*判断obj在左还是在右*/
442 				if( tipItemLeft < tipItemWrapWidth/2 - tipItemWidth/2 ){
443 					left = tipItemLeft + tipItemWidth + tipArrowW;
444 					/*判断obj在上还是在下*/
445 					if( tipItemTop < tipItemWrapHeight/2 - tipItemHeight/2 ){
446 						top = tipItemTop;
447 						tipArrow.addClass('l-tipArrow-leftTop');
448 					}else{
449 						top = tipItemTop + tipItemHeight - tipWrapHeight;
450 						tipArrow.addClass('l-tipArrow-leftBottom');
451 					}
452 				}else{
453 					left = tipItemLeft - tipItemWidth -tipArrowW;
454 					/*判断obj在上还是在下*/

456 						top = tipItemTop;
457 						tipArrow.addClass('l-tipArrow-rightTop');
458 					}else{
459 						top = tipItemTop + tipItemHeight - tipWrapHeight;
460 						tipArrow.addClass('l-tipArrow-rightBottom');
461 					}
462 				}
463 			}

465 		},
466 		
467 		/*删除*/

475 	if( isInitShow ){
476 		tipFn.position();
477 	}else{
478 		tipItem.live(event,function(){
479 			if( isArrow ){
480 				tipFn.arrow();
481 			}else if( isTrack ){
482 				tipFn.track();
483 			}else{
484 				tipFn.defaultPositon();
485 			}
486 		}).live('mousemove',function(){
487 			if( isTrack ){
488 				tipFn.track();
489 			}
490 		}).live('mouseout',function(){
491 			tipFn.remove();
492 		});
493 		
494 	}//end if
495 	
496 };
