弹出框

普通的：点击打开

						$('#pop').click(function(){
							zUI.ui.pop.open({
								title:'标题',
								html:'111111'
							});
						});

带按钮的：点击打开

						$('#pop').click(function(){
							zUI.ui.pop.open({
								title:'标题',
								html:'111111',
								btns:[
									{
										text:'确定',
										onClick:function(){
											alert('确定');
										}
									},
									{}
								]
							});
						});

支持ajax：

						$('#pop').click(function(){
							zUI.ui.pop.open({
								title:'标题',
								ajax:'http://127.0.0.1'
								btns:[
									{
										text:'确定',
										onClick:function(){
											alert('确定');
										}
									},
									{}
								]
							});
						});
